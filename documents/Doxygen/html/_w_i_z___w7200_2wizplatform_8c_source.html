394 void RCC_Configuration(void)
395 {
396  ErrorStatus HSEStartUpStatus;
397 
398  RCC_DeInit(); /* RCC system reset(for debug purpose) */
399  RCC_HSEConfig(RCC_HSE_ON); /* Enable HSE */
400  HSEStartUpStatus = RCC_WaitForHSEStartUp(); /* Wait till HSE is ready */
401 
402  if(HSEStartUpStatus == SUCCESS)
403  {
404  FLASH_PrefetchBufferCmd(FLASH_PrefetchBuffer_Enable); /* Enable Prefetch Buffer */
405  FLASH_SetLatency(FLASH_Latency_2); /* Flash 2 wait state */
406  RCC_HCLKConfig(RCC_SYSCLK_Div1); /* HCLK = SYSCLK */
407  RCC_PCLK2Config(RCC_HCLK_Div1); /* PCLK2 = HCLK */
408  RCC_PCLK1Config(RCC_HCLK_Div2); /* PCLK1 = HCLK/2 */
409  RCC_PLLConfig(RCC_PLLSource_HSE_Div1, RCC_PLLMul_9); /* PLLCLK = 8MHz * 9 = 72 MHz */
410  RCC_PLLCmd(ENABLE); /* Enable PLL */
411 
412  while(RCC_GetFlagStatus(RCC_FLAG_PLLRDY) == RESET) {} /* Wait till PLL is ready */
413  RCC_SYSCLKConfig(RCC_SYSCLKSource_PLLCLK); /* Select PLL as system clock source */
414  while(RCC_GetSYSCLKSource() != 0x08) {} /* Wait till PLL is used as system clock source */
415  }

417  RCC_APB2PeriphClockCmd(RCC_APB2Periph_GPIOA, ENABLE);
418  RCC_APB2PeriphClockCmd(RCC_APB2Periph_GPIOB, ENABLE);
419  RCC_APB2PeriphClockCmd(RCC_APB2Periph_GPIOC, ENABLE);
