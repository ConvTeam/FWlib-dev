335 void RCC_Configuration(void)
336 {
337  ErrorStatus HSEStartUpStatus;
338 
339  RCC_DeInit(); /* RCC system reset(for debug purpose) */
340  RCC_HSEConfig(RCC_HSE_ON); /* Enable HSE */
341  HSEStartUpStatus = RCC_WaitForHSEStartUp(); /* Wait till HSE is ready */
342 
343  if(HSEStartUpStatus == SUCCESS)
344  {
345  FLASH_PrefetchBufferCmd(FLASH_PrefetchBuffer_Enable); /* Enable Prefetch Buffer */
346  FLASH_SetLatency(FLASH_Latency_2); /* Flash 2 wait state */
347  RCC_HCLKConfig(RCC_SYSCLK_Div1); /* HCLK = SYSCLK */
348  RCC_PCLK2Config(RCC_HCLK_Div1); /* PCLK2 = HCLK */
349  RCC_PCLK1Config(RCC_HCLK_Div2); /* PCLK1 = HCLK/2 */
350  RCC_PLLConfig(RCC_PLLSource_HSE_Div1, RCC_PLLMul_9); /* PLLCLK = 8MHz * 9 = 72 MHz */
351  RCC_PLLCmd(ENABLE); /* Enable PLL */
352 
353  while(RCC_GetFlagStatus(RCC_FLAG_PLLRDY) == RESET) {} /* Wait till PLL is ready */
354  RCC_SYSCLKConfig(RCC_SYSCLKSource_PLLCLK); /* Select PLL as system clock source */
355  while(RCC_GetSYSCLKSource() != 0x08) {} /* Wait till PLL is used as system clock source */
356  }

358  RCC_APB2PeriphClockCmd(RCC_APB2Periph_GPIOA, ENABLE);
359  RCC_APB2PeriphClockCmd(RCC_APB2Periph_GPIOB, ENABLE);
360  RCC_APB2PeriphClockCmd(RCC_APB2Periph_GPIOC, ENABLE);
